# Car Scraper Cron Jobs（run_scheduler.sh 使用）
# cron からは run_scheduler.sh だけを叩く方式に統一
#
# 事前準備:
#   1. プロジェクトルートで: chmod +x run_scheduler.sh
#   2. 下の BASE を、実際のプロジェクトパスに合わせる
#      - シンボリックリンクを使う場合: /Users/yutaka/car_scraper_link
#      - 実パスを使う場合: /Users/yutaka/Library/CloudStorage/GoogleDrive-.../car_scraper

SHELL=/bin/bash
PATH=/Users/yutaka/.pyenv/shims:/usr/local/bin:/usr/bin:/bin
MAILTO=""

# プロジェクトのパス（シンボリックリンク or 実パス、どちらか一方を有効に）
BASE=/Users/yutaka/car_scraper_link

# Weekly execution (Sundays at 3:00 AM)
# 週次実行（日曜日 午前3時）。CRON_INVOKED=1 でログに [CRON] と出し判別可能に
0 3 * * 0 CRON_INVOKED=1 $BASE/run_scheduler.sh >> $BASE/logs/cron.log 2>&1

# Health check (daily at 6:00 AM)
# ヘルスチェック（毎日 午前6時）
0 6 * * * cd $BASE && /Users/yutaka/.pyenv/shims/python3 scheduled_scraper.py --health-check >> $BASE/logs/health_check.log 2>&1

# 別曜日に「選択していないメーカー」だけ実行（第1引数で設定ファイルを指定）
# 例: 水曜 3:00 に config/carmaker_selection_other.txt のメーカーで実行
# 0 3 * * 3 CRON_INVOKED=1 $BASE/run_scheduler.sh config/carmaker_selection_other.txt >> $BASE/logs/cron_other.log 2>&1
